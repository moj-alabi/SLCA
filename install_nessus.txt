#!/bin/bash

# Simple Nessus installer for Kali Linux / Debian-based systems

echo "[*] Updating system..."
sudo apt update -y

echo "[*] Downloading the latest Nessus .deb package..."
# You can change the URL if needed; this points to the official Tenable downloads.
wget -O nessus.deb https://www.tenable.com/downloads/api/v1/public/pages/nessus/downloads/17838/download?i_agree_to_tenable_license_agreement=true

echo "[*] Installing Nessus..."
sudo dpkg -i nessus.deb

echo "[*] Fixing missing dependencies..."
sudo apt --fix-broken install -y

echo "[*] Enabling and starting Nessus service..."
sudo systemctl enable nessusd
sudo systemctl start nessusd

echo
echo "==========================================="
echo " Nessus installation complete!"

echo " ðŸ‘‰ https://<your-ip>:8834/"
echo "==========================================="
