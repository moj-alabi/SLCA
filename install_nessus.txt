#!/bin/bash

# Update package list
sudo apt update

# Download Nessus package
curl --request GET \
  --url 'https://www.tenable.com/downloads/api/v2/pages/nessus/files/Nessus-10.11.0-ubuntu1604_amd64.deb' \
  --output 'Nessus-10.11.0-ubuntu1604_amd64.deb'

# Install Nessus
sudo dpkg -i Nessus-10.11.0-ubuntu1604_amd64.deb

# Enable Nessus service at startup
sudo systemctl enable nessusd.service

# Start Nessus service
sudo systemctl start nessusd.service

echo "Nessus installation complete!"
echo "Access the web UI at: https://<your-ip>:8834"
